Data

df <- readRDS("./data/metadata/df.RDS")

Supplemental figure: Correlation matrix of MIC

MIC_correlation <-  cor(df %>% select("MERO_log_2","MVB_log_2","IMI_log_2","IR_log_2","CZA_log_2","TGC_log_2","CST_log_2","AMK_log_2","PLZ_log_2") , method="spearman") %>% round(.,2)
 
drugs <-  rownames(MIC_correlation) %>% gsub("_log_2","",.)

corr_heatmap <- ComplexHeatmap::pheatmap(MIC_correlation %>% as.matrix,border_color = "black",name ="Spearman's Rank Correlation Coefficient",legend=F,fontsize = 14,
    row_names_max_width = max_text_width(rownames(MIC_correlation), gp = gpar(fontsize = 16)),
    column_names_max_height = max_text_width(rownames(MIC_correlation), gp = gpar(fontsize = 16)),
    cluster_rows=T,cluster_cols = T,labels_row = drugs,labels_col=drugs,number_color = "black",breaks = c(-1,0,1),color=c("blue","white","red"),display_numbers = T)  

## Complex heatmap legend 
fig_legend <- Legend(at = c(-1, -0.5, 0, 0.5, 1),labels = c("-1.00", "-0.50", "0.00", "0.50", "1.00"),title = "Spearman's Rank Correlation Coefficient",direction = "horizontal",col_fun = colorRamp2(breaks = c(-1,  0,  1),colors = c("blue","white","red")), title_gp = gpar(fontsize = 16),labels_gp=gpar(fontsize=16),legend_width = unit(14, "cm"))
 
# Print heatmap
fig_heatmap_image <- grid.grabExpr(draw(corr_heatmap))
fig_legend_image <- grid.grabExpr(draw(fig_legend))

s_figure_2 <-plot_grid(fig_heatmap_image,NULL,fig_legend_image,ncol=1,axis="l", align="hv",rel_heights = c(1,-.08,.25))
